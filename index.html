  function showSection(id) {

    const sections = document.querySelectorAll('.section');

    sections.forEach(section => section.classList.remove('active'));

    document.getElementById(id).classList.add('active');

  }


  const API_KEY = 'AIzaSyCZBabW2LPVQ710RFc7gqxjWba83Xknojs';

  const SHEET_ID = '1lwU1ffiQooHS5B6pJieRABOGraY_ITjnlIavcpLwetQ';

  const SHEET_NAME = 'Seafarers';


  async function verifyCertificate() {

    const query = document.getElementById('searchInput').value.toLowerCase().trim();

    const resultDiv = document.getElementById('result');

    resultDiv.innerHTML = 'Verifying...';


    const url = `https://sheets.googleapis.com/v4/spreadsheets/${SHEET_ID}/values/${SHEET_NAME}?key=${API_KEY}`;


    try {

      const res = await fetch(url);

      const data = await res.json();

      const rows = data.values;


      if (!rows || rows.length < 2) {

        resultDiv.innerHTML = 'No data found.';

        return;

      }


      const headers = rows[0];

      const matches = rows.slice(1).filter(row =>

        row.some(cell => (cell || '').toLowerCase().includes(query))

      );


      if (matches.length === 0) {

        resultDiv.innerHTML = 'No matching certificate found.';

        return;

      }


      let html = '<table><tr>';

      headers.forEach(header => {

        html += `<th>${header}</th>`;

      });

      html += '</tr>';


      matches.forEach(row => {

        html += '<tr>';

        headers.forEach((_, i) => {

          html += `<td>${row[i] || ''}</td>`;

        });

        html += '</tr>';

      });


      html += '</table>';

      resultDiv.innerHTML = html;


    } catch (err) {

      console.error(err);

      resultDiv.innerHTML = 'Error retrieving data.';

    }

  }
